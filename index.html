Project Overview

Problem
Structural Safety
Manual inspection of concrete surfaces is slow, inconsistent, and expensive. Cracks signal structural degradation and must be caught early.
Solution
YOLOv8 Vision
YOLOv8-seg detects crack bounding boxes, classifies type, and produces pixel-level masks — all in a single forward pass at 30+ FPS.
Output
Web Dashboard
Upload image → get annotated bounding boxes, crack type label, confidence score, segmentation mask, and severity report.
Novelty
Detection + Severity
Not just detection — classify crack as hairline, structural, fatigue, or spalling AND compute a 1–5 severity score from mask area.

2. Training — 3 Lines with Ultralytics

PYTHON — train.py
from ultralytics import YOLO

# Load pretrained YOLOv8 segmentation model (COCO weights)
model = YOLO('yolov8s-seg.pt')   # n=nano, s=small, m=medium

# Train — Ultralytics handles augmentation, LR schedule, logging
results = model.train(
    data='crack_dataset.yaml',
    epochs=100,
    imgsz=640,
    batch=16,
    device='0',              # GPU; use 'cpu' if no GPU
    patience=20,             # early stopping
    save=True,
    project='crack_runs',
    name='yolov8s_seg_v1',
    exist_ok=True,
    plots=True,              # saves confusion matrix, PR curve
    val=True
)

# Results auto-saved to crack_runs/yolov8s_seg_v1/
# Best weights at: crack_runs/yolov8s_seg_v1/weights/best.pt
print(f"mAP50: {results.results_dict['metrics/mAP50(B)']:.4f}")
print(f"mAP50-95: {results.results_dict['metrics/mAP50-95(B)']:.4f}")

3. Inference + Severity Scoring

PYTHON — inference.py
from ultralytics import YOLO
import cv2, numpy as np

CLASS_NAMES = ['No Crack', 'Hairline', 'Structural', 'Fatigue', 'Spalling']

model = YOLO('crack_runs/yolov8s_seg_v1/weights/best.pt')

def compute_severity(mask_pixels, img_pixels):
    ratio = mask_pixels / img_pixels
    if ratio < 0.01:   return 1
    elif ratio < 0.03: return 2
    elif ratio < 0.07: return 3
    elif ratio < 0.15: return 4
    else:               return 5

def predict_image(image_path):
    results = model.predict(image_path, conf=0.4, iou=0.5)[0]
    img = cv2.imread(image_path)
    H, W = img.shape[:2]
    detections = []

    for i, box in enumerate(results.boxes):
        cls_id   = int(box.cls[0])
        conf     = float(box.conf[0])
        mask     = results.masks.data[i].numpy() if results.masks else None
        severity = compute_severity(mask.sum(), H*W) if mask is not None else 0

        detections.append({
            'crack_type':  CLASS_NAMES[cls_id],
            'confidence':  round(conf * 100, 2),
            'severity':    severity,
            'bbox':        box.xyxy[0].tolist()
        })

    # Save annotated image
    annotated = results.plot()   # draws boxes + masks automatically
    cv2.imwrite('output_annotated.jpg', annotated)
    return detections, annotated

4. FastAPI Endpoint

PYTHON — api.py
from fastapi import FastAPI, UploadFile, File
from fastapi.middleware.cors import CORSMiddleware
import io, base64, cv2, numpy as np
from PIL import Image
from ultralytics import YOLO
from inference import predict_image, compute_severity

app = FastAPI(title="Crack Detection API", version="2.0")
app.add_middleware(CORSMiddleware, allow_origins=["*"])

model = YOLO("best.pt")   # loaded once at startup

@app.post("/predict")
async def predict(file: UploadFile = File(...)):
    contents = await file.read()
    nparr = np.frombuffer(contents, np.uint8)
    img   = cv2.imdecode(nparr, cv2.IMREAD_COLOR)

    results  = model.predict(img, conf=0.4)[0]
    annotated = results.plot()   # boxes + masks drawn
    _, buffer  = cv2.imencode('.jpg', annotated)
    img_b64    = base64.b64encode(buffer).decode()

    detections = []
    H, W = img.shape[:2]
    for i, box in enumerate(results.boxes):
        mask     = results.masks.data[i].numpy() if results.masks else None
        severity = compute_severity(mask.sum(), H*W) if mask is not None else 0
        detections.append({
            "crack_type": CLASS_NAMES[int(box.cls[0])],
            "confidence": round(float(box.conf[0]) * 100, 2),
            "severity":   severity,
            "bbox":       box.xyxy[0].tolist()
        })

    return {
        "crack_detected": len(detections) > 0,
        "detections":     detections,
        "annotated_image": img_b64
    }
