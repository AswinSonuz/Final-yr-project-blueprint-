Project Overview

Structural Safety
Manual inspection of concrete surfaces is slow, inconsistent, and expensive. Cracks signal structural degradation and must be caught early.

Deep Learning Vision
A CNN-based image classification + segmentation pipeline that detects crack type, location, and severity from camera photos in real time.

Upload image → get crack type, confidence score, annotated heatmap, and severity report. Works on mobile camera too.
Novelty
Classification + Severity
Not just detection — classify crack as hairline, structural, fatigue, or corrosion-induced and give a 1–5 severity score.

1. Dataset & DataLoader

PYTHON — dataset.py
import torch
from torch.utils.data import Dataset, DataLoader
from torchvision import transforms
from PIL import Image
import os, glob

class CrackDataset(Dataset):
    def __init__(self, root_dir, split='train', transform=None):
        self.image_paths = []
        self.labels = []
        self.transform = transform
        
        # Expects: root/train/crack/*.jpg and root/train/no_crack/*.jpg
        for label, cls in enumerate(['no_crack', 'crack']):
            paths = glob.glob(f"{root_dir}/{split}/{cls}/*.jpg")
            self.image_paths.extend(paths)
            self.labels.extend([label] * len(paths))

    def __len__(self):
        return len(self.image_paths)

    def __getitem__(self, idx):
        img = Image.open(self.image_paths[idx]).convert('RGB')
        if self.transform:
            img = self.transform(img)
        return img, self.labels[idx]

# Transforms with augmentation for training
train_transform = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.RandomHorizontalFlip(),
    transforms.RandomRotation(15),
    transforms.ColorJitter(brightness=0.3, contrast=0.3),
    transforms.ToTensor(),
    transforms.Normalize([0.485, 0.456, 0.406],
                             [0.229, 0.224, 0.225])
])

2. Model — Transfer Learning with ResNet-50

PYTHON — model.py
import torch
import torch.nn as nn
from torchvision.models import resnet50, ResNet50_Weights

class CrackClassifier(nn.Module):
    def __init__(self, num_classes=5, freeze_backbone=True):
        super().__init__()
        self.backbone = resnet50(weights=ResNet50_Weights.DEFAULT)
        
        if freeze_backbone:
            for p in self.backbone.parameters():
                p.requires_grad = False
        
        # Replace final FC layer
        in_features = self.backbone.fc.in_features
        self.backbone.fc = nn.Sequential(
            nn.Dropout(0.4),
            nn.Linear(in_features, 256),
            nn.ReLU(),
            nn.Dropout(0.3),
            nn.Linear(256, num_classes)
        )

    def forward(self, x):
        return self.backbone(x)

# Classes: 0=NoCrack 1=Hairline 2=Structural 3=Fatigue 4=Spalling
CLASS_NAMES = ['No Crack', 'Hairline', 'Structural', 'Fatigue', 'Spalling']

3. Training Loop

PYTHON — train.py
import torch, mlflow
from torch.optim.lr_scheduler import ReduceLROnPlateau

def train_model(model, train_loader, val_loader, epochs=30):
    device = torch.device('cuda' if torch.cuda.is_available() else 'cpu')
    model.to(device)
    
    optimizer = torch.optim.Adam(
        filter(lambda p: p.requires_grad, model.parameters()),
        lr=1e-4, weight_decay=1e-5
    )
    criterion = nn.CrossEntropyLoss()
    scheduler = ReduceLROnPlateau(optimizer, patience=3, factor=0.5)
    best_val_acc = 0
    
    with mlflow.start_run():
        for epoch in range(epochs):
            model.train()
            train_loss = 0
            
            for imgs, labels in train_loader:
                imgs, labels = imgs.to(device), labels.to(device)
                optimizer.zero_grad()
                outputs = model(imgs)
                loss = criterion(outputs, labels)
                loss.backward()
                optimizer.step()
                train_loss += loss.item()
            
            val_acc = evaluate(model, val_loader, device)
            scheduler.step(train_loss)
            mlflow.log_metrics({'val_acc': val_acc, 'loss': train_loss}, step=epoch)

                best_val_acc = val_acc
                torch.save(model.state_dict(), 'best_model.pt')

PYTHON — api.py
from fastapi import FastAPI, UploadFile, File
from fastapi.middleware.cors import CORSMiddleware
import torch, io, base64
from PIL import Image
from model import CrackClassifier, CLASS_NAMES
from gradcam import generate_gradcam

app = FastAPI(title="Crack Detection API", version="1.0")
app.add_middleware(CORSMiddleware, allow_origins=["*"])

# Load model once at startup
model = CrackClassifier(num_classes=5)
model.load_state_dict(torch.load("best_model.pt", map_location="cpu"))
model.eval()

@app.post("/predict")
async def predict(file: UploadFile = File(...)):
    contents = await file.read()
    img = Image.open(io.BytesIO(contents)).convert("RGB")
    
    tensor = preprocess(img).unsqueeze(0)
    
    with torch.no_grad():
        logits = model(tensor)
        probs = torch.nn.functional.softmax(logits, dim=1)
        confidence, pred_class = probs.max(1)
    
    cam_b64 = generate_gradcam(model, tensor, pred_class)
    severity = compute_severity(pred_class.item(), confidence.item())
    
    return {
        "crack_detected": pred_class.item() > 0,
        "crack_type": CLASS_NAMES[pred_class.item()],
        "confidence": round(confidence.item() * 100, 2),
        "severity_score": severity,
        "gradcam_image": cam_b64
    }
